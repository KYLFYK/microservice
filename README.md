## App Cleverdeus

MONGO_DSN=mongodb://mongobase:27017/cleverdeusdb
MONGO_DATABASE=cleverdeusdb
MONGO_USER=
MONGO_PASSWORD=

MONGO_ROOT_PASSWORD=

USER_SERVICE_PORT=4001
USER_SERVICE_HOST=127.0.0.1
CUSTOMER_SERVICE_PORT=4002
